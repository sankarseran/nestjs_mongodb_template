  async checkRoom(params: object): Promise<ICommunication[]> {
    console.log("Hi Coming here", params);
    return await this.communicationModel.find(
      {
        $or: [
          {
            from_user_id: params["from_user_id"],

  async findUsersList(params: object): Promise<ICommunication[]> {
    // return await this.communicationModel.find(
    //   { $or:[ { from_user_id : ID }, { to_user_id : ID } ] },
    // ).exec();
    console.log("Params", Object.keys(params));
    console.log("Params", params["user_id"]);
 
    let id = params["user_id"];
    let limit = parseInt(params["limit"]);
    let offset = parseInt(params["offset"]);
 
    let match_query = {
      $or: [
        { from_user_id: mongoose.Types.ObjectId(id) },
        { to_user_id: mongoose.Types.ObjectId(id) }
      ]
    };
 
    var aggregate = [
      {
        $lookup: {
          from: "user",
          localField: "to_user_id",
          foreignField: "_id",
          as: "user"
        }
      },
      {
        $unwind: {
          path: "$user",
          preserveNullAndEmptyArrays: true
        }
      },
      { $match: match_query },
      { $sort: { created_date: 1 } },
      {
        $group: {
          //'_id': { 'from_user_id' : '$from_user_id', 'to_user_id' : '$to_user_id' },
          _id: "$to_user_id",
          date: { $last: "$created_date" },
          username: { $last: "$user.username" },
          message: { $last: "$message" },
          from_user_id: { $last: "$from_user_id" },
          to_user_id: { $last: "$to_user_id" },
          created_date: { $last: "$created_date" },
          communication_id: { $last: "$_id" }
        }
      },
      {
        $project: {
          _id: "$communication_id",
          username: "$username",
          from_user_id: "$from_user_id",
          to_user_id: "$to_user_id",
          date: "$created_date",
          message: "$message"
 
          //'list' : '$list'
        }
      },
      { $skip: offset },
      { $limit: limit }
    ];
 
    return await this.communicationModel.aggregate(aggregate).exec();
  }
