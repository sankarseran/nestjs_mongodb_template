  @Get("/:room_id/:user_id")
  public async getTodo(
    @Response() res,
    @Param() viewCommunicationDto: ViewCommunicationDto
  ) {
    const results = await this.CommunicationService.find({
      room_id: viewCommunicationDto.room_id,
      is_blocked: false
    });
    results.map(key => {
      key.is_read_flag = false;
 
      if (key.is_read.indexOf(viewCommunicationDto.user_id) >= 0) {
        key.is_read_flag = true;
      }
      return key;
    });
    
    const userStatus = await this.CommunicationRoomService.findOne({
      _id: viewCommunicationDto.room_id,
    });
    let is_blocked = false;
    if(userStatus.is_blocked.length > 1) {
      is_blocked = true;
    } else if(userStatus.is_blocked.length > 0) {
      if(userStatus.is_blocked.indexOf(viewCommunicationDto.user_id) < 0){
        is_blocked = true;
      }
    }
 
    this.utility.sendSuccess(res, results, "succ", is_blocked);
  }
