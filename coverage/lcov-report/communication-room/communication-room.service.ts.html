import { debug } from "console";

import { ResMessage } from ".././common/res.message";
 
@Injectable()

  constructor(

  async checkRoom(params: object): Promise<ICommunicationRoom[]> {
    
    let data = params["to_user_id"];
    data.push(params["from_user_id"]);
    
    return await this.communicationModel.find(
      { users: { $size: data.length, $all: data }},
      { _id: 1,is_blocked: 1 }
    );
  }
  
  async checkConversation(params: object): Promise<ICommunicationRoom[]> {
    
    let data = params["to_user_id"];
    data.push(params["from_user_id"]);

      { users: { $size: data.length, $all: data },"is_delete":{"$nin":[params["from_user_id"]]}},

  async updateDelete(params: object): Promise<ICommunicationRoom[]> {
    
    let result = await this.communicationModel.find({"_id":params['room_id']});
     result.map(key => {
        params['to_user_id'].map(key1 => {
                 key.is_delete.splice(key.is_delete.indexOf(mongoose.Types.ObjectId(key1)), 1);
         })
         
     key.save();
     
    });
    
    return result;
  }
 
  async findUsersList(params: object): Promise<ICommunicationRoom[]> {
    
    //console.log(params);
    let id = params['user_id'];
    let limit = parseInt(params['limit']);
    let offset = parseInt(params['offset']);
    
    let from_date = (typeof params['from_date'] != 'undefined') ? params['from_date'] : '';
    let to_date = (typeof params['to_date'] != 'undefined') ? params['to_date'] : '';
    
    
    let match_query : any;
 
    match_query = { 
      users :  { $in : [mongoose.Types.ObjectId(id)] },
    }
 
    //For Search
    let keyword = (typeof params['keyword'] != 'undefined') ? params['keyword'] : '';
    let match_query_search : any = {};
    let keyword_1 : string = "";
    let keyword_lowercase : string = "";
    let sort : any = {};
    
 
    if(from_date != "" && to_date != "")  {
      sort = {
        "created_date": -1
      }
      match_query = {
        "users" :  { $in : [mongoose.Types.ObjectId(id)] },
        "communications.created_date" : { '$gte': new Date(from_date),'$lte': new Date(to_date) }
      };
    } else if(keyword != '') {
      keyword = new RegExp(params['keyword'], 'i');
      keyword_1 = params['keyword'];
      keyword_lowercase = params['keyword'].toLowerCase();
      match_query_search = {
        $or:[ 
          {
            'communications.user_name' : {
              $regex: keyword
            },
          },
          {
            'communications.communications.type_value.message' : {
              $regex: keyword
            }
          }
          
        ] 
      }
      sort = {
        "sort" : 1
      }
      
    } else {
      sort = {
        "created_date": -1
      }
      match_query = {
        "users" :  { $in : [mongoose.Types.ObjectId(id)] }
      };
    }
    
    match_query['is_delete'] = { $nin : [mongoose.Types.ObjectId(id)] };
    
    
    var aggregate = [
 
       {
          '$lookup': {
              'from': 'communications',
              'localField': '_id',
              'foreignField': 'room_id',
              'as': 'communications'
          }
      },
      
      {
          '$unwind': {
              "path": '$communications',
              "preserveNullAndEmptyArrays": true
          }
      },
      {
            '$lookup': {
                'from': 'user',
                'localField': 'users',
                'foreignField': '_id',
                'as': 'user'
            }
        },
        {
          '$unwind': {
              "path": '$user',
              "preserveNullAndEmptyArrays": true
          }
        },
      { '$match': match_query },
      { "$sort": { "communications.created_date": 1 } },
      { 
        "$group": { 
          "_id": "$_id",
          "communications" : { 
            $push:  { 
              "user_name": { $toLower: "$user.username" },
              "communications" : "$communications",
              "user_id" : "$user._id",
              "is_blocked": "$is_blocked"
        }
          }
        }
      },
      {
        '$unwind': {
            "path": '$communications',
            "preserveNullAndEmptyArrays": false
        }
      },
      {
        '$match': {
          'communications.user_id': { $ne : mongoose.Types.ObjectId(id) },
          'communications.communications.is_blocked': false
        } 
      },
      { 
        '$match': match_query_search 
      },
      { 
        "$group": { 
          "_id": "$_id",
          "communications" : { $last : "$communications" }
        }
      },
      {
        $project: {
          _id: "$_id",
          //username: "$username",
          user_name : '$communications.user_name',
          user_name_id : '$communications.user_id',
          from_user_id: "$communications.communications.from_user_id",
          to_user_id: "$communications.communications.to_user_id",
          date: "$communications.communications.created_date",
          communications_id: "$communications.communications._id",
          type: "$communications.communications.type",
          type_value: "$communications.communications.type_value",
          message: "$communications.communications.type_value",
          is_forward: "$communications.communications.is_forward",
          created_date: "$communications.communications.created_date",
          updated_date: "$communications.communications.updated_date",
          is_blocked: "$communications.is_blocked",
          'sort': {
            $cond: { 
              if: { $eq: [ keyword, "" ] }, 
              then: "1",
              else: {
                $cond: {
                  if: { 
                    $gte: [ { $indexOfBytes: [ "$communications.user_name", keyword_lowercase ] }, 0 ] 
                  }, then: 1, else: 2 
                }
              }
            }
          },
          is_read_flag:{
            $cond: { if: { $gte: [ { $indexOfArray: [ "$communications.communications.is_read", mongoose.Types.ObjectId( params["user_id"]) ] }, 0 ] }, then: true, else: false }
          },
               
        }
      },
      { "$sort": sort },
      { '$skip': offset },
      { '$limit': limit },
 
  ];
 
  return await this.communicationModel.aggregate(aggregate).exec()
 
  }
