  Controller,
  Get,
  Response,
  HttpStatus,

import { CommunicationRoomService } from "./communication-room.service";
import { CommunicationRoomDto } from "./dto/communication-room.dto";
import { ApiUseTags, ApiResponse, ApiBearerAuth } from "@nestjs/swagger";
import { Utility } from ".././common/utility";
const utility = new Utility();
import { ResMessage } from ".././common/res.message";
@ApiUseTags("Communication Room")
@Controller("communication-room")
export class CommunicationRoomController {
  constructor(
    private readonly CommunicationRoomService: CommunicationRoomService
  ) {}
 
  @Get()
  public async getCommunicationList(@Response() res) {
    const todos = await this.CommunicationRoomService.findAll();
    return res.status(HttpStatus.OK).json(todos);
  }
 
  @Get("find")
  public async findTodo(@Response() res, @Body() body) {
    const queryCondition = body;
    const todos = await this.CommunicationRoomService.findOne(queryCondition);
    return res.status(HttpStatus.OK).json(todos);
  }
 
  @Get("/:id")
  public async getTodo(@Response() res, @Param() param) {
    const todos = await this.CommunicationRoomService.findById(param.id);
    return res.status(HttpStatus.OK).json(todos);
  }
 
  @Get("/list/:id/:limit/:offset")
  public async getCommunicationUsersList(
    @Response() res,
    @Param() CommunicationRoomDto: CommunicationRoomDto
  ) {
    const todos = await this.CommunicationRoomService.findUsersList(

    return res.status(HttpStatus.OK).json(todos);
  }

  @Patch("/:id")
  public async updateTodo(@Param() param, @Response() res, @Body() body) {
    const todo = await this.CommunicationRoomService.update(param.id, body);
